CREATE TABLE Users (
id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
username CHAR(64) NOT NULL UNIQUE,
password CHAR(64) NOT NULL,
first varbinary(200),
last varbinary(200),
created TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
lastlogin TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE Contacts (
id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
user VARCHAR(7),
data varbinary(15000)
);

CREATE TABLE Tags (
id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
user VARCHAR(7),
name varbinary(100)
);

// this has to be part of the contact json
CREATE TABLE Images (
id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
user VARCHAR(7),
contact VARCHAR(7),
picture varbinary(65000)
);

CREATE TABLE Passwords (
id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
user VARCHAR(7),
data varbinary(4000)
);
